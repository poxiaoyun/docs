## Overview

The Moha SDK is compatible with the Hugging Face Transformers and Datasets libraries, allowing models and datasets to be loaded directly from the Moha Hub, just like using the official Hub.

## Model loading - Transformers

### Basic usage

```python
from transformers import AutoModel, AutoTokenizer
from xiaoshiai_hub import snapshot_download

# 推荐方式：先下载，再加载（最稳定）
model_path = snapshot_download("ai-lab/qwen2-7b")
model = AutoModel.from_pretrained(model_path)
tokenizer = AutoTokenizer.from_pretrained(model_path)
```

### Configure Moha endpoint

```python
import os
from transformers import AutoModel
from xiaoshiai_hub import snapshot_download

# 如果你的运行环境需要自定义地址，可设置环境变量
os.environ["HF_ENDPOINT"] = "https://moha.xiaoshiai.com"
os.environ["MOHA_TOKEN"] = "sk_xxx..."

# 仍建议先通过 SDK 下载，再交给 Transformers 加载
model_path = snapshot_download("ai-lab/qwen2-7b")
model = AutoModel.from_pretrained(model_path)
```

## Complete workflow example

### Scenario 1: Fine-tuning - Loading model and data from Hub

```python
import torch
from transformers import (
    AutoModel, 
    AutoTokenizer, 
    Trainer, 
    TrainingArguments
)
from datasets import load_dataset

# 1. 加载预训练模型和 tokenizer
model_name = "ai-lab/qwen2-7b"
model = AutoModel.from_pretrained(model_name)
tokenizer = AutoTokenizer.from_pretrained(model_name)

# 2. 加载数据集
dataset = load_dataset("ai-lab/finetuning-data")

# 3. 数据预处理
def preprocess_function(examples):
    return tokenizer(
        examples["text"],
        truncation=True,
        max_length=512
    )

tokenized_dataset = dataset.map(
    preprocess_function,
    batched=True,
    remove_columns=dataset["train"].column_names
)

# 4. 配置训练参数
training_args = TrainingArguments(
    output_dir="./results",
    num_train_epochs=3,
    per_device_train_batch_size=8,
    save_steps=100,
    save_total_limit=2,
)

# 5. 创建训练器
trainer = Trainer(
    model=model,
    args=training_args,
    train_dataset=tokenized_dataset["train"],
    eval_dataset=tokenized_dataset["validation"],
)

# 6. 开始训练
trainer.train()

# 7. 保存微调模型
model.save_pretrained("./my-finetuned-model")
tokenizer.save_pretrained("./my-finetuned-model")
```

### Scenario 2: Evaluation - Evaluate the model on the Hub dataset

```python
from transformers import pipeline
from datasets import load_dataset
from sklearn.metrics import accuracy_score, f1_score

# 1. 加载预训练管道
classifier = pipeline(
    "text-classification",
    model="ai-lab/sentiment-bert"
)

# 2. 加载评测数据集
eval_dataset = load_dataset("ai-lab/sentiment-data", split="test")

# 3. 批量预测
predictions = []
labels = []

for example in eval_dataset:
    # 预测
    result = classifier(example["text"])
    pred_label = result[0]["label"]
    predictions.append(pred_label)
    labels.append(example["label"])

# 4. 评测
accuracy = accuracy_score(labels, predictions)
f1 = f1_score(labels, predictions, average="weighted")

print(f"Accuracy: {accuracy:.4f}")
print(f"F1-Score: {f1:.4f}")
```

### Scenario 3: Inference - Direct Inference Application

```python
from transformers import AutoModelForCausalLM, AutoTokenizer
import torch

# 1. 加载模型和 tokenizer
model_name = "ai-lab/qwen2-7b"
model = AutoModelForCausalLM.from_pretrained(
    model_name,
    device_map="auto",
    torch_dtype=torch.float16
)
tokenizer = AutoTokenizer.from_pretrained(model_name)

# 2. 创建推理函数
def generate_text(prompt, max_length=100):
    inputs = tokenizer(prompt, return_tensors="pt").to(model.device)
    outputs = model.generate(
        **inputs,
        max_length=max_length,
        temperature=0.7,
        top_p=0.9,
        do_sample=True
    )
    return tokenizer.decode(outputs[0], skip_special_tokens=True)

# 3. 推理
result = generate_text("Python 是")
print(result)
```

### Scenario 4: Data processing pipeline

```python
from datasets import load_dataset
from transformers import AutoTokenizer

# 1. 加载数据集
raw_dataset = load_dataset("ai-lab/raw-text-data")

# 2. 加载 tokenizer
tokenizer = AutoTokenizer.from_pretrained("ai-lab/qwen2-7b")

# 3. 清洗和处理
def clean_text(example):
    text = example["text"].lower().strip()
    return {"text": text}

cleaned = raw_dataset.map(clean_text)

# 4. Token 化
def tokenize_function(examples):
    return tokenizer(
        examples["text"],
        truncation=True,
        max_length=512,
        padding="max_length"
    )

tokenized = cleaned.map(
    tokenize_function,
    batched=True,
    remove_columns=["text"]
)

# 5. 创建数据加载器
from torch.utils.data import DataLoader

dataloader = DataLoader(
    tokenized["train"],
    batch_size=32,
    shuffle=True
)

# 6. 迭代数据
for batch in dataloader:
    print(f"Batch size: {len(batch)}")
    # 用于训练
    break
```

## Multi-GPU inference

### Distributed reasoning

```python
import torch
from transformers import AutoModelForCausalLM, AutoTokenizer

# 单 GPU（使用 device_map）
model = AutoModelForCausalLM.from_pretrained(
    "ai-lab/qwen2-7b",
    device_map="auto",  # 自动分配到可用 GPU
)

# 多 GPU（使用 DataParallel）
if torch.cuda.device_count() > 1:
    model = torch.nn.DataParallel(model)

# 推理
prompt = "Tell me a story"
inputs = tokenizer(prompt, return_tensors="pt").to("cuda:0")
outputs = model.generate(**inputs)
print(tokenizer.decode(outputs[0]))
```

### Quantitative acceleration

```python
from transformers import AutoModelForCausalLM, AutoTokenizer
from transformers import BitsAndBytesConfig

# INT8 量化
quantization_config = BitsAndBytesConfig(
    load_in_8bit=True
)

model = AutoModelForCausalLM.from_pretrained(
    "ai-lab/qwen2-7b",
    quantization_config=quantization_config
)

# INT4 量化
quantization_config = BitsAndBytesConfig(
    load_in_4bit=True,
    bnb_4bit_compute_dtype=torch.float16
)

model = AutoModelForCausalLM.from_pretrained(
    "ai-lab/qwen2-7b",
    quantization_config=quantization_config
)
```

## Performance optimization

### Batch processing

```python
from transformers import AutoTokenizer, AutoModel
import torch

model = AutoModel.from_pretrained("ai-lab/qwen2-7b")
tokenizer = AutoTokenizer.from_pretrained("ai-lab/qwen2-7b")

texts = [
    "Hello world",
    "Good morning",
    "How are you"
]

# 批处理
inputs = tokenizer(
    texts,
    return_tensors="pt",
    padding=True,
    truncation=True,
    max_length=512
)

with torch.no_grad():
    outputs = model(**inputs)
```

### Model caching

```python
from functools import lru_cache
from transformers import AutoModel

@lru_cache(maxsize=1)
def load_model():
    return AutoModel.from_pretrained("ai-lab/qwen2-7b")

# 第一次调用加载模型
model = load_model()

# 后续调用使用缓存
model = load_model()  # 直接返回缓存的模型
```
