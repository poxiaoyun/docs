## 训练工作流概述

完整的模型训练流程包括四个阶段：

```
准备 → 训练 → 评测 → 上传
```

## 第一步：准备数据和模型

### 加载预训练模型

```python
from transformers import (
    AutoModel,
    AutoTokenizer,
    AutoModelForCausalLM
```

### 预处理数据

```python
from transformers import AutoTokenizer
from datasets import load_dataset

tokenizer = AutoTokenizer.from_pretrained("ai-lab/qwen2-7b")
dataset = load_dataset("ai-lab/training-data")

# 数据预处理函数
def preprocess_function(examples):
    result = tokenizer(
        examples["text"],
        truncation=True,
        max_length=512,
        padding="max_length"
    )
    
    # 准备标签（用于 causal language modeling）
    result["labels"] = result["input_ids"].copy()
    
    return result

# 应用预处理
tokenized_dataset = dataset.map(
    preprocess_function,
    batched=True,
    num_proc=4,  # 使用 4 个进程并行处理
    remove_columns=["text"]
)

print(f"处理后的数据集: {tokenized_dataset}")
```

## 第二步：配置训练参数

### 基础训练配置

```python
from transformers import TrainingArguments

# 创建训练参数
training_args = TrainingArguments(
    output_dir="./output",
    num_train_epochs=3,
    per_device_train_batch_size=8,
    per_device_eval_batch_size=8,
    
    # 优化参数
    learning_rate=5e-5,
    weight_decay=0.01,
    warmup_ratio=0.1,
    
    # 评测和保存
    evaluation_strategy="steps",
    eval_steps=100,
    save_strategy="steps",
    save_steps=100,
    save_total_limit=3,
    
    # 日志
    logging_steps=10,
    logging_dir='./logs',
    
    # 硬件优化
    fp16=True,  # 混合精度训练
    gradient_accumulation_steps=4,
)

print(training_args)
```

### 高级训练配置

```python
from transformers import TrainingArguments

training_args = TrainingArguments(
    output_dir="./output",
    num_train_epochs=3,
    per_device_train_batch_size=8,
    
    # LoRA 适配器配置（用于大模型微调）
    peft_type="LORA",
    r=8,
    lora_alpha=32,
    lora_dropout=0.1,
    
    # 梯度检查点（节省内存）
    gradient_checkpointing=True,
    
    # 学习率调度
    lr_scheduler_type="cosine",
    
    # 分布式训练
    ddp_find_unused_parameters=False,
    
    # 推送到 Hub
    push_to_hub=True,
    hub_model_id="my-org/finetuned-model",
    hub_strategy="checkpoint",  # 保存最佳检查点
)
```

## 第三步：开始训练

### 标准微调

```python
from transformers import (
    Trainer,
    TrainingArguments,
    AutoModelForCausalLM,
    AutoTokenizer
)
from datasets import load_dataset

# 1. 加载模型和数据
model = AutoModelForCausalLM.from_pretrained("ai-lab/qwen2-7b")
tokenizer = AutoTokenizer.from_pretrained("ai-lab/qwen2-7b")
dataset = load_dataset("ai-lab/training-data")

# 2. 预处理
def preprocess(examples):
    return tokenizer(
        examples["text"],
        truncation=True,
        max_length=512,
        padding="max_length"
    )

dataset = dataset.map(preprocess, batched=True)

# 3. 配置训练
training_args = TrainingArguments(
    output_dir="./model",
    num_train_epochs=3,
    per_device_train_batch_size=8,
    learning_rate=5e-5,
)

# 4. 创建训练器
trainer = Trainer(
    model=model,
    args=training_args,
    train_dataset=dataset["train"],
    eval_dataset=dataset["validation"],
)

# 5. 开始训练
trainer.train()

print("✓ 训练完成")
```

### 使用 LoRA 的高效微调

```python
import torch
from peft import get_peft_model, LoraConfig, TaskType
from transformers import (
    AutoModelForCausalLM,
    AutoTokenizer,
    Trainer,
    TrainingArguments
)
from datasets import load_dataset

# 1. 加载基础模型
base_model = AutoModelForCausalLM.from_pretrained(
    "ai-lab/qwen2-7b",
    device_map="auto"
)
tokenizer = AutoTokenizer.from_pretrained("ai-lab/qwen2-7b")

# 2. 配置 LoRA
peft_config = LoraConfig(
    task_type=TaskType.CAUSAL_LM,
    r=8,                    # LoRA 秩
    lora_alpha=32,          # 缩放因子
    lora_dropout=0.1,       # Dropout 率
    bias="none",
    target_modules=["q_proj", "v_proj"],  # 应用于哪些模块
)

# 3. 应用 LoRA
model = get_peft_model(base_model, peft_config)
model.print_trainable_parameters()
# trainable params: 8,388,608 || all params: 7,255,834,624 || trainable%: 0.12

# 4. 加载数据
dataset = load_dataset("ai-lab/training-data")

# 5. 配置训练
training_args = TrainingArguments(
    output_dir="./lora_model",
    num_train_epochs=3,
    per_device_train_batch_size=4,  # 更小的批量（内存不足）
    learning_rate=1e-4,
    fp16=True,
)

# 6. 训练
trainer = Trainer(
    model=model,
    args=training_args,
    train_dataset=dataset["train"],
)
trainer.train()

print("✓ LoRA 微调完成")
```

## 第四步：评测模型

### 在验证集上评测

```python
from transformers import Trainer, TrainingArguments, AutoModelForCausalLM, AutoTokenizer
from datasets import load_dataset
import torch

# 加载训练好的模型
model = AutoModelForCausalLM.from_pretrained("./model")

# 加载评测数据
eval_dataset = load_dataset("ai-lab/eval-data", split="test")

# 创建评测器
trainer = Trainer(
    model=model,
    args=TrainingArguments(output_dir="./eval"),
    eval_dataset=eval_dataset,
)

# 运行评测
results = trainer.evaluate()
print(results)
# 输出: {'eval_loss': 2.345, 'eval_runtime': 123.45, ...}
```

### 自定义评测指标

```python
from datasets import load_dataset
from transformers import pipeline
import numpy as np

# 加载分类模型
classifier = pipeline(
    "text-classification",
    model="./finetuned-model"
)

# 加载评测数据
test_data = load_dataset("ai-lab/test-data", split="test")

# 自定义评测
def compute_metrics(predictions, labels):
    acc = sum(p == l for p, l in zip(predictions, labels)) / len(labels)
    return {"accuracy": acc}

# 批量预测
predictions = []
labels = []
for example in test_data:
    result = classifier(example["text"])
    pred = result[0]["label"]
    predictions.append(pred)
    labels.append(example["label"])

# 计算指标
metrics = compute_metrics(predictions, labels)
print(f"准确率: {metrics['accuracy']:.4f}")
```

## 第五步：导出和保存

### 保存微调模型

```python
from transformers import AutoModelForCausalLM, AutoTokenizer

# 加载训练好的模型
model = AutoModelForCausalLM.from_pretrained("./output/checkpoint-1000")
tokenizer = AutoTokenizer.from_pretrained("ai-lab/qwen2-7b")

# 保存到本地
model.save_pretrained("./my_finetuned_model")
tokenizer.save_pretrained("./my_finetuned_model")

print("✓ 模型已保存到 ./my_finetuned_model")
```

### 导出 ONNX 格式

```python
import torch
from transformers import AutoModelForCausalLM, AutoTokenizer

# 加载模型
model = AutoModelForCausalLM.from_pretrained("./my_model")
tokenizer = AutoTokenizer.from_pretrained("./my_model")

# 导出 ONNX
from transformers.onnx import export

export(
    preprocessor=tokenizer,
    model=model,
    config_class=model.config,
    output="./model_onnx",
)

print("✓ 模型已导出为 ONNX 格式")
```

### 导出 TorchScript

```python
import torch
from transformers import AutoModelForCausalLM, AutoTokenizer

# 加载模型
model = AutoModelForCausalLM.from_pretrained("./my_model")
model.eval()

# 创建示例输入
tokenizer = AutoTokenizer.from_pretrained("./my_model")
example_input = tokenizer.encode("Hello", return_tensors="pt")

# 转换为 TorchScript
traced_model = torch.jit.trace(model, example_input)
traced_model.save("./model.pt")

print("✓ 模型已导出为 TorchScript 格式")
```

## 完整端到端示例

```python
from datasets import load_dataset
from transformers import (
    AutoModelForCausalLM,
    AutoTokenizer,
    Trainer,
    TrainingArguments
)
from xiaoshiai_hub import upload_folder

# 1. 准备
print("1️⃣ 准备数据和模型...")
model = AutoModelForCausalLM.from_pretrained("ai-lab/qwen2-7b")
tokenizer = AutoTokenizer.from_pretrained("ai-lab/qwen2-7b")
dataset = load_dataset("ai-lab/training-data")

# 预处理
def preprocess(examples):
    return tokenizer(
        examples["text"],
        truncation=True,
        max_length=512
    )

dataset = dataset.map(preprocess, batched=True, remove_columns=["text"])

# 2. 训练
print("2️⃣ 开始训练...")
training_args = TrainingArguments(
    output_dir="./final_model",
    num_train_epochs=3,
    per_device_train_batch_size=8,
    learning_rate=5e-5,
    save_strategy="steps",
    save_steps=100,
)

trainer = Trainer(
    model=model,
    args=training_args,
    train_dataset=dataset["train"],
)
trainer.train()

# 3. 保存
print("3️⃣ 保存模型...")
model.save_pretrained("./final_model")
tokenizer.save_pretrained("./final_model")

# 4. 上传
print("4️⃣ 上传到 Moha Hub...")
upload_folder(
    folder_path="./final_model",
    repo_id="my-org/my-finetuned-model",
    commit_message="Complete finetuning workflow"
)

print("✅ 工作流完成！")
```

## 性能优化技巧

### 内存优化

```python
from transformers import TrainingArguments

training_args = TrainingArguments(
    output_dir="./output",
    
    # 梯度检查点（节省 ~30% 内存）
    gradient_checkpointing=True,
    
    # 梯度累积（降低内存需求）
    gradient_accumulation_steps=4,
    
    # 混合精度（节省 ~50% 内存）
    fp16=True,
    
    # 减小批量大小
    per_device_train_batch_size=4,
)
```

### 速度优化

```python
from transformers import TrainingArguments

training_args = TrainingArguments(
```

### Q: 如何保存最佳模型？

```python
from transformers import TrainingArguments

training_args = TrainingArguments(
    output_dir="./output",
    save_strategy="steps",
    save_steps=100,
    save_total_limit=1,  # 只保存最佳 1 个检查点
    load_best_model_at_end=True,
    metric_for_best_model="eval_loss",
)
```

### Q: 如何恢复中断的训练？

```python
from transformers import Trainer, TrainingArguments

trainer = Trainer(
    model=model,
    args=TrainingArguments(
        output_dir="./output",
        resume_from_checkpoint=True,  # 自动检测并恢复
        ...
    ),
    ...
)
trainer.train()
```
